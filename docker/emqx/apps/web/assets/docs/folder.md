# 目录结构

```txt
├── .vscode                         # vscode 项目配置
├── api                             # Web 请求
├── assets                          # 静态资源
|   |── demo                        # 示例
│   ├── icons                       # 图标文件
|   ├── docs                        # Markdown 文档
|   └── logo.svg                    # logo
├── components                      # Vue 通用组件
├── config                          # 配置文件
├── layout                          # 布局组件
├── lib                             # 三方组件
├── locale                          # 国际化
├── mixins                          # Vue 混入
├── mock                            # 接口模拟
├── router                          # 路由配置
├── store                           # 状态管理
├── styles                          # 样式
├── utils                           # 工具库
├── views                           # 页面视图
├── .eslintrc.json                  # JavaScript 代码检查
├── .editorconfig                   # 编辑器配置
├── .gitignore                      # git 上传过滤
├── .gitattributes                  # git 文本配置
├── .markdownlint.json              # Markdown 代码检查
├── .prettierrc.json                # prettier 格式化配置
├── .stylelintrc.js                 # CSS 代码检查
├── app.js                          # app 组件
├── favicon.ico                     # 浏览器图标
├── index.html                      # web 入口
├── jsconfig.json                   # jsconfig 配置
├── main.css                        # css 入口
└── main.js                         # js 入口
```
